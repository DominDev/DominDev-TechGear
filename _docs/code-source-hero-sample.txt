    <script>
        /* --- 4. RAIN ENGINE (Vanilla JS) --- */
        document.addEventListener('DOMContentLoaded', () => {
            const canvas = document.getElementById('rain-canvas');
            const ctx = canvas.getContext('2d');
            
            let width, height;
            let drops = [];
            
            // Konfiguracja deszczu
            const config = {
                count: 150,        // Ilość kropel
                color: '#80e5ff',  // Kolor (lekki cyan)
                speedBase: 15,     // Prędkość
                lengthBase: 20     // Długość kropli
            };

            function resize() {
                width = window.innerWidth;
                height = window.innerHeight;
                canvas.width = width;
                canvas.height = height;
            }

            class Drop {
                constructor() {
                    this.reset(true);
                }

                reset(initial = false) {
                    this.x = Math.random() * width;
                    // Jeśli start, losuj pozycję Y na ekranie, inaczej resetuj nad ekranem
                    this.y = initial ? Math.random() * height : -50;
                    this.z = Math.random() * 0.5 + 0.5; // Głębia (wpływa na wielkość i prędkość)
                    this.len = (Math.random() * 20 + config.lengthBase) * this.z;
                    this.speed = (Math.random() * 10 + config.speedBase) * this.z;
                    this.opacity = Math.random() * 0.4 + 0.1;
                }

                update() {
                    this.y += this.speed;
                    if (this.y > height) {
                        this.reset();
                    }
                }

                draw() {
                    ctx.beginPath();
                    ctx.strokeStyle = `rgba(128, 229, 255, ${this.opacity})`;
                    ctx.lineWidth = 1.5 * this.z;
                    ctx.moveTo(this.x, this.y);
                    ctx.lineTo(this.x, this.y + this.len);
                    ctx.stroke();
                }
            }

            function initRain() {
                drops = [];
                for(let i=0; i<config.count; i++) {
                    drops.push(new Drop());
                }
            }

            function animate() {
                // Clear with slight trail effect optional, but here we want crisp rain
                ctx.clearRect(0, 0, width, height);

                drops.forEach(drop => {
                    drop.update();
                    drop.draw();
                });

                requestAnimationFrame(animate);
            }

            // Init
            window.addEventListener('resize', () => {
                resize();
                // Opcjonalnie reset kropel przy zmianie rozmiaru
            });
            
            resize();
            initRain();
            animate();
        });
    </script>
