            { id: 5, name: 'CYBERDECK MK.IV', category: 'keyboard', price: 1299, img: 'https://images.unsplash.com/photo-1587829741301-dc798b91a91e?auto=format&fit=crop&q=80&w=600' },
            { id: 6, name: 'VIPER OPTICAL', category: 'mouse', price: 299, img: 'https://images.unsplash.com/photo-1527864550417-7fd91fc51a46?auto=format&fit=crop&q=80&w=600' }
        ];

        /* --------------------------------------------------------
           LOGIC: CART SYSTEM
        -------------------------------------------------------- */
        let cart = []; // Array of objects: { id, qty, ...product }

        function toggleCart() {
            document.getElementById('cartSidebar').classList.toggle('open');
            document.getElementById('overlayBg').classList.toggle('open');
        }

        // Add to Cart with Grouping Logic
        function addToCart(id) {
            const existingItem = cart.find(item => item.id === id);

            if (existingItem) {
                existingItem.qty++;
            } else {
                const product = products.find(p => p.id === id);
                cart.push({ ...product, qty: 1 });
            }
            updateCartUI();
            toggleCart(); // Open cart to show feedback
        }

        // Modify Quantity
        function changeQty(id, change) {
            const itemIndex = cart.findIndex(item => item.id === id);
            if (itemIndex > -1) {
                cart[itemIndex].qty += change;
                
                // Remove if qty <= 0
                if (cart[itemIndex].qty <= 0) {
                    cart.splice(itemIndex, 1);
                }
            }
            updateCartUI();
        }

        // Remove Item Completely
        function removeItem(id) {
            cart = cart.filter(item => item.id !== id);
            updateCartUI();
        }

        function updateCartUI() {
            const container = document.getElementById('cartItemsContainer');
            const totalEl = document.getElementById('cartTotalValue');
            const countEl = document.getElementById('cartCount');
            
            container.innerHTML = '';
            let grandTotal = 0;
            let totalCount = 0;

            if (cart.length === 0) {
                container.innerHTML = '<div style="text-align: center; margin-top: 4rem; color: #444; font-family: var(--font-code);">[ INVENTORY_EMPTY ]</div>';
            } else {
                cart.forEach(item => {
                    grandTotal += item.price * item.qty;
                    totalCount += item.qty;

                    const row = document.createElement('div');
                    row.className = 'cart-item-row';
                    row.innerHTML = `
                        <img src="${item.img}" class="cart-img" alt="item">
                        <div class="cart-info">
                            <div class="cart-name">${item.name}</div>
                            <div class="cart-price">${item.price} PLN <span style="color:#666; font-size:0.8rem;">x ${item.qty}</span></div>
                        </div>
                        <div class="cart-controls">
                            <button class="qty-btn" onclick="changeQty(${item.id}, -1)">-</button>
                            <span style="font-family: var(--font-code); color: var(--color-accent);">${item.qty}</span>
                            <button class="qty-btn" onclick="changeQty(${item.id}, 1)">+</button>
                            <button class="qty-btn" style="margin-left:5px; color:#ff3333;" onclick="removeItem(${item.id})">×</button>
                        </div>
                    `;
                    container.appendChild(row);
                });
            }

            totalEl.innerText = grandTotal.toFixed(2) + ' PLN';
            countEl.innerText = totalCount;
            countEl.style.display = totalCount > 0 ? 'block' : 'none';
        }

        /* --------------------------------------------------------
           LOGIC: RENDER & FILTER
        -------------------------------------------------------- */
        const grid = document.getElementById('productGrid');
        const filters = document.querySelectorAll('.filter-tab');

        function renderProducts(cat = 'all') {
            grid.innerHTML = '';
            
            const filtered = cat === 'all' ? products : products.filter(p => p.category === cat);
            
            filtered.forEach((p, index) => {
                const card = document.createElement('div');
                card.className = 'card reveal';
                // Staggered animation delay
                card.style.transitionDelay = `${index * 0.1}s`;

                card.innerHTML = `
                    <div class="product-img-wrap">
                        <img src="${p.img}" class="product-img" alt="${p.name}" loading="lazy">
                    </div>
                    <div>
                        <div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:0.5rem;">
                            <span class="text-code" style="color:#666; font-size:0.75rem;">// ITEM_${p.id}</span>
                            <span class="text-code text-accent">${p.price} PLN</span>
                        </div>
                        <h3 style="font-size:1.1rem; margin-bottom:1.5rem;">${p.name}</h3>
                        <button class="btn" style="width:100%; font-size:0.9rem;" onclick="addToCart(${p.id})">ADD TO INV</button>
                    </div>
                `;
                grid.appendChild(card);
            });

            // Trigger animation
            setTimeout(() => {
                document.querySelectorAll('.card.reveal').forEach(el => el.classList.add('active'));
            }, 50);
        }

        // Init
        renderProducts();

        // Filters
        filters.forEach(btn => {
            btn.addEventListener('click', () => {
                filters.forEach(f => f.classList.remove('active'));
                btn.classList.add('active');
                renderProducts(btn.dataset.filter);
            });
        });

        /* --------------------------------------------------------
           LOGIC: SCROLL OBSERVER
        -------------------------------------------------------- */
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('active');
                }
            });
        }, { threshold: 0.1 });

        document.querySelectorAll('.reveal').forEach(el => observer.observe(el));
